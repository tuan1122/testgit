<template>
  <el-dialog
    v-model="localVisible"
    title="Upload files"
    width="520px"
    @closed="resetFiles"
  >
    <!-- Drag Upload -->
    <el-upload
      class="compact-upload"
      drag
      multiple
      :auto-upload="false"
      :file-list="fileList"
      :show-file-list="false"
      :on-change="handleChange"
      accept=".png,.jpg,.jpeg,.pdf"
    >
      <el-icon><upload-filled /></el-icon>
      <div class="hint">Kéo thả hoặc click để chọn ảnh / PDF</div>
    </el-upload>

    <!-- File List -->
    <div class="file-list">
      <div v-for="file in fileList" :key="file.uid" class="file-item">
        <div class="file-left">
          <el-icon class="file-icon">
            <Document v-if="isPdf(file)" />
            <Picture v-else />
          </el-icon>
          <span class="file-name">{{ file.name }}</span>
        </div>

        <div class="file-actions">
          <el-button link type="primary" @click="openPreview(file)">
            <el-icon><View /></el-icon>
          </el-button>

          <el-button link type="danger" @click="removeFile(file)">
            <el-icon><Delete /></el-icon>
          </el-button>
        </div>
      </div>
    </div>

    <!-- Footer -->
    <template #footer>
      <el-button @click="localVisible = false">Đóng</el-button>
      <el-button type="primary" @click="submitUpload">Upload</el-button>
    </template>

    <!-- Preview Dialog -->
    <el-dialog v-model="previewVisible" width="80%" top="5vh">
      <div class="preview-filename">{{ previewName }}</div>

      <img
        v-if="previewType === 'image'"
        :src="previewUrl"
        class="preview-image"
      />
      <iframe v-else :src="previewUrl" class="preview-pdf"></iframe>
    </el-dialog>
  </el-dialog>
</template>

<script setup>
import { ref, watch } from 'vue';
import {
  UploadFilled,
  View,
  Delete,
  Document,
  Picture,
} from '@element-plus/icons-vue';

/* ===== v-model:isVisible ===== */
const props = defineProps({
  isVisible: Boolean,
});
const emit = defineEmits(['update:isVisible']);

const localVisible = ref(props.isVisible);

watch(
  () => props.isVisible,
  (v) => (localVisible.value = v)
);
watch(localVisible, (v) => emit('update:isVisible', v));

/* ===== File list ===== */
const fileList = ref([]);

const previewVisible = ref(false);
const previewUrl = ref('');
const previewType = ref('image');
const previewName = ref('');

function handleChange(file, files) {
  fileList.value = files;
}

function removeFile(file) {
  fileList.value = fileList.value.filter((f) => f.uid !== file.uid);
}

function isPdf(file) {
  return file.name.toLowerCase().endsWith('.pdf');
}

function openPreview(file) {
  previewUrl.value = URL.createObjectURL(file.raw);
  previewType.value = isPdf(file) ? 'pdf' : 'image';
  previewName.value = file.name;
  previewVisible.value = true;
}

function submitUpload() {
  console.log('Files ready:', fileList.value);
  localVisible.value = false;
}

/* ===== Reset when dialog fully closed ===== */
function resetFiles() {
  fileList.value = [];
  previewVisible.value = false;
  previewUrl.value = '';
  previewName.value = '';
}
</script>
